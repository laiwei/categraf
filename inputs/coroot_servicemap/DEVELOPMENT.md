#### TCP 连接跟踪示例 (tcp/state.c)

```c
#include "vmlinux.h"
#include <bpf/bpf_helpers.h>
#include <bpf/bpf_tracing.h>

struct tcp_event {
    __u64 fd;
    __u64 timestamp;
    __u32 type;
    __u32 pid;
    __u16 sport;
    __u16 dport;
    __u8 saddr[16];
    __u8 daddr[16];
};

struct {
    __uint(type, BPF_MAP_TYPE_PERF_EVENT_ARRAY);
    __uint(key_size, sizeof(int));
    __uint(value_size, sizeof(int));
} tcp_events SEC(".maps");

SEC("tracepoint/sock/inet_sock_set_state")
int trace_inet_sock_set_state(struct trace_event_raw_inet_sock_set_state *ctx) {
    if (ctx->protocol != IPPROTO_TCP) {
        return 0;
    }

    struct tcp_event event = {};
    event.timestamp = bpf_ktime_get_ns();
    event.pid = bpf_get_current_pid_tgid() >> 32;
    // ... 填充其他字段

    bpf_perf_event_output(ctx, &tcp_events, BPF_F_CURRENT_CPU, &event, sizeof(event));
    return 0;
}

char _license[] SEC("license") = "GPL";
```

### 2. 编译 eBPF 程序

创建编译脚本 `inputs/coroot_servicemap/tracer/ebpf/build.sh`:

```bash
#!/bin/bash
set -e

EBPF_DIR="$(cd "$(dirname "$0")" && pwd)"
cd "$EBPF_DIR"

# 检查依赖
if ! command -v clang &> /dev/null; then
    echo "Error: clang not found"
    exit 1
fi

if ! command -v llvm-strip &> /dev/null; then
    echo "Error: llvm-strip not found"
    exit 1
fi

# 编译不同内核版本的程序
for KERNEL in 416 420 506 512; do
    for ARCH in x86 arm64; do
        echo "Building for kernel ${KERNEL} ${ARCH}..."
        
        clang -g -O2 -target bpf \
            -D__KERNEL_FROM=${KERNEL} \
            -D__TARGET_ARCH_${ARCH} \
            -c ebpf.c -o ebpf${KERNEL}${ARCH}.o
        
        llvm-strip --strip-debug ebpf${KERNEL}${ARCH}.o
    done
done

# 生成 Go 嵌入代码
echo "Generating Go code..."
./generate_go.sh
```

### 4. 更新 Tracer 加载代码

修改 `tracer.go` 中的 `loadEBPF()` 方法：

```go
func (t *Tracer) loadEBPF() error {
    // 获取内核版本
    kv, err := getKernelVersion()
    if err != nil {
        return err
    }

    // 选择合适的程序
    progs, ok := ebpfProgs[runtime.GOARCH]
    if !ok {
        return fmt.Errorf("unsupported architecture: %s", runtime.GOARCH)
    }

    var prog []byte
    for _, p := range progs {
        pv, _ := parseVersion(p.version)
        if kv.GreaterOrEqual(pv) {
            prog = p.prog
            break
        }
    }

    if len(prog) == 0 {
        return fmt.Errorf("no suitable eBPF program for kernel %s", kv)
    }

    // 解压程序
    progBytes, err := decompressEBPFProgram(prog)
    if err != nil {
        return err
    }

    // 加载到内核
    spec, err := ebpf.LoadCollectionSpecFromReader(bytes.NewReader(progBytes))
    if err != nil {
        return err
    }

    t.collection, err = ebpf.NewCollection(spec)
    if err != nil {
        return err
    }

    // 附加到内核钩子
    return t.attachProbes()
}

func (t *Tracer) attachProbes() error {
    // 附加 tracepoint
    for name, prog := range t.collection.Programs {
        switch {
        case strings.HasPrefix(name, "trace_inet_sock_set_state"):
            l, err := link.Tracepoint("sock", "inet_sock_set_state", prog, nil)
            if err != nil {
                return err
            }
            t.links = append(t.links, l)
        // ... 其他探针
        }
    }

    // 启动事件读取器
    return t.startReaders()
}
```

func (r *Registry) getContainerIDByPID(pid uint32) string {
    // 读取 /proc/<pid>/cgroup
    data, err := os.ReadFile(fmt.Sprintf("/proc/%d/cgroup", pid))
    if err != nil {
        return ""
    }

    // 解析 cgroup 路径
    // Docker: /docker/<container-id>

## 编译和运行

### 本地开发

```bash
# 编译 eBPF 程序
cd inputs/coroot_servicemap/tracer/ebpf
./build.sh

# 编译 Categraf
cd ../../../../
make build

# 运行（需要 root）
sudo ./categraf --configs conf
```

### Docker 构建

```dockerfile
FROM golang:1.21 as builder

# 安装 eBPF 工具链
RUN apt-get update && apt-get install -y \
    clang \
    llvm \
    libbpf-dev \
    linux-headers-generic

WORKDIR /build
COPY . .

# 编译 eBPF 程序
RUN cd inputs/coroot_servicemap/tracer/ebpf && ./build.sh

# 编译 Categraf
RUN make build

FROM ubuntu:22.04
COPY --from=builder /build/categraf /usr/bin/
CMD ["/usr/bin/categraf"]
```
